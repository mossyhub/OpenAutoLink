cmake_minimum_required(VERSION 3.16)
project(ocd_headless_test CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -fPIC")

find_package(Boost REQUIRED COMPONENTS system log)
find_package(Protobuf REQUIRED)
find_package(OpenSSL REQUIRED)

include_directories(
    ${OCD_AASDK_INCLUDE}
    ${OCD_AASDK_PROTO_INCLUDE}
    ${Boost_INCLUDE_DIRS}
    ${PROTOBUF_INCLUDE_DIR}
    ${OPENSSL_INCLUDE_DIR}
)

add_definitions(-DBOOST_ALL_DYN_LINK)

add_executable(ocd_headless_test ocd_headless_test.cpp)

target_link_libraries(ocd_headless_test
    ${OCD_AASDK_LIB}
    ${OCD_AAP_PROTO_LIB}
    ${Boost_LIBRARIES}
    ${PROTOBUF_LIBRARIES}
    ${OPENSSL_LIBRARIES}
    pthread
)
