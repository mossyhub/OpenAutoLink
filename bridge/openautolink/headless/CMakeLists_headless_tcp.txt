cmake_minimum_required(VERSION 3.10)
project(headless_tcp_server CXX)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -fPIC")
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake_modules/")

find_package(Boost REQUIRED COMPONENTS system log)
find_package(Protobuf REQUIRED)
find_package(OpenSSL REQUIRED)

# aasdk paths (set via -D flags)
# AASDK_INCLUDE_DIRS, AASDK_LIBRARIES, AASDK_PROTO_INCLUDE_DIRS, AASDK_PROTO_LIBRARIES

include_directories(
    ${AASDK_PROTO_INCLUDE_DIRS}
    ${AASDK_INCLUDE_DIRS}
    ${Boost_INCLUDE_DIRS}
    ${PROTOBUF_INCLUDE_DIR}
    ${OPENSSL_INCLUDE_DIR}
)

add_definitions(-DBOOST_ALL_DYN_LINK)

add_executable(headless_tcp_server headless_tcp_server.cpp)

target_link_libraries(headless_tcp_server
    ${AASDK_LIBRARIES}
    ${AASDK_PROTO_LIBRARIES}
    ${Boost_LIBRARIES}
    ${PROTOBUF_LIBRARIES}
    ${OPENSSL_LIBRARIES}
    pthread
)
